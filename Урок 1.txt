#2. Пользователь вводит время в секундах. 
# Переведите время в часы, минуты и секунды и выведите в формате чч:мм:сс. Используйте форматирование строк.

time = int(input("Введите время в секундах "))
hours = time // 3600
minutes = (time - hours * 3600) // 60
seconds = time - (hours * 3600 + minutes * 60)
print(f"Время в формате чч:мм:сс   {hours} : {minutes} : {seconds}")

# Найдите самую большую цифру в числе. Для решения используйте цикл while и арифметические операции.

n = int(input("Введите число >>> "))

#5. Запросите у пользователя значения выручки и издержек фирмы. 
# Определите, с каким финансовым результатом работает фирма (прибыль — выручка больше издержек, или убыток — издержки больше выручки). 
# Выведите соответствующее сообщение. Если фирма отработала с прибылью, вычислите рентабельность выручки (соотношение прибыли к выручке). 
# Далее запросите численность сотрудников фирмы и определите прибыль фирмы в расчете на одного сотрудника.

revenue = int(input("Выручка >>> "))
costs = int(input("Издержки >>> "))
profit = revenue - costs
employees = int(input("Количество сотрудников >>> "))
salary = profit / employees
rent = profit / revenue
while True:
    if revenue > costs:
        print(f"Выручка больше издержек! Поздравляем!: {profit:.2f}")
        print(f"Соотношение прибыли к выручке: {rent:.2f}")
        print(f"Прибыль фирмы в расчете на одного сотрудника: {salary:.2f}")
        break
    else:
        print(f"Издержки больше выручки: {profit:.2f}, :( ")
        break
